CQ01.sparql
PREFIX edintinm: <http://vocab.linkeddata.es/datosabiertos/def/oferta-inmobiliaria/>
PREFIX skos: <http://www.w3.org/2004/02/skos/core#>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>

# INM01: What types of real estate transactions exist in the listings?
# SPANISH: ¿Qué tipos de transacción inmobiliaria existen en los anuncios?

SELECT DISTINCT ?transactionType ?transactionLabel ?transactionLabelEs
WHERE {
  ?listing a edintinm:RealEstateListing ;
           edintinm:transactionType ?transactionType .
  
  OPTIONAL {
    ?transactionType rdfs:label ?transactionLabel .
    FILTER (lang(?transactionLabel) = "en")
  }
  
  OPTIONAL {
    ?transactionType rdfs:label ?transactionLabelEs .
    FILTER (lang(?transactionLabelEs) = "es")
  }
}

ORDER BY ?transactionLabel
CQ02.sparql
PREFIX edintinm: <http://vocab.linkeddata.es/datosabiertos/def/oferta-inmobiliaria/>
PREFIX dcterms: <http://purl.org/dc/terms/>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>

# INM02: What information is contained in a real estate listing?
# SPANISH: ¿Qué información contiene un anuncio inmobiliario?

SELECT ?listing ?listingId ?isActive ?propertyTypeLabelEs ?transactionTypeLabelEs
WHERE {
  ?listing a edintinm:RealEstateListing ;
           dcterms:identifier ?listingId ;
           edintinm:isActive ?isActive .
  
  OPTIONAL {
    ?listing edintinm:propertyType ?propertyType .
    ?propertyType rdfs:label ?propertyTypeLabelEs .
    FILTER (lang(?propertyTypeLabelEs) = "es")
  }
  
  OPTIONAL {
    ?listing edintinm:transactionType ?transactionType .
    ?transactionType rdfs:label ?transactionTypeLabelEs .
    FILTER (lang(?transactionTypeLabelEs) = "es")
  }
}

ORDER BY ?listingId
CQ03.sparql
PREFIX edintinm: <http://vocab.linkeddata.es/datosabiertos/def/oferta-inmobiliaria/>
PREFIX dcterms: <http://purl.org/dc/terms/>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>

# INM03: How is a listing related to a cadastral entity?
# SPANISH: ¿Cómo se relaciona un anuncio con una entidad catastral?

SELECT ?listing ?listingId ?cadastralEntity ?entityType ?entityLabelEs
WHERE {
  ?listing a edintinm:RealEstateListing ;
           dcterms:identifier ?listingId ;
           edintinm:refersToCadastralEntity ?cadastralEntity .
  
  ?cadastralEntity rdfs:label ?entityLabelEs .
  FILTER (lang(?entityLabelEs) = "es")
  
  BIND(
    IF(STRENDS(STR(?cadastralEntity), "CadastralParcel"), 
       "Parcela Catastral",
       IF(STRENDS(STR(?cadastralEntity), "CadastralProperty"),
          "Inmueble Catastral",
          "Entidad Desconocida")
    ) AS ?entityType
  )
}

ORDER BY ?listingId
CQ04.sparql
PREFIX edintinm: <http://vocab.linkeddata.es/datosabiertos/def/oferta-inmobiliaria/>
PREFIX skos: <http://www.w3.org/2004/02/skos/core#>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>

# INM04: What properties does a property have in a sale listing?
# SPANISH: ¿Qué propiedades tiene un inmueble en venta?

SELECT DISTINCT ?listing ?listingId ?propertyType ?propertyLabelEs ?propertyTypeLabelEs
WHERE {
  ?listing a edintinm:RealEstateListing ;
           edintinm:propertyType ?propertyType ;
           edintinm:transactionType ?transactionType .
  
  ?transactionType skos:prefLabel ?transactionLabel .
  FILTER (STRLCASE(STR(?transactionLabel)) = "venta" || 
         STRLCASE(STR(?transactionLabel)) = "sale")
  
  OPTIONAL {
    ?propertyType rdfs:label ?propertyLabelEs .
    FILTER (lang(?propertyLabelEs) = "es")
  }
  
  OPTIONAL {
    ?propertyType skos:prefLabel ?propertyTypeLabelEs .
    FILTER (lang(?propertyTypeLabelEs) = "es")
  }
}

ORDER BY ?listingId
CQ05.sparql
PREFIX edintinm: <http://vocab.linkeddata.es/datosabiertos/def/oferta-inmobiliaria/>
PREFIX dcterms: <http://purl.org/dc/terms/>

# INM05: What is the current status of a real estate listing?
# SPANISH: ¿Cuál es el estado actual de un anuncio inmobiliario?

SELECT ?listing ?listingId ?isActive ?statusLabel
WHERE {
  ?listing a edintinm:RealEstateListing ;
           dcterms:identifier ?listingId ;
           edintinm:isActive ?isActive .
  
  BIND(
    IF(?isActive = true, 
       "Activo"^^<http://www.w3.org/2001/XMLSchema#string>, 
       "Inactivo"^^<http://www.w3.org/2001/XMLSchema#string>)
    AS ?statusLabel)
}

ORDER BY ?listingId

CQ07.sparql
PREFIX edintinm: <http://vocab.linkeddata.es/datosabiertos/def/oferta-inmobiliaria/>
PREFIX dcterms: <http://purl.org/dc/terms/>
PREFIX geo: <http://www.opengis.net/ont/geosparql#>
PREFIX catastro: <https://edint.github.io/edint-ontologia-catastro/ontology/catastro#>

# INM07: How is the location of a property specified?
# SPANISH: ¿Cómo se especifica la ubicación de un inmueble?

SELECT ?listing ?listingId ?cadastralEntity ?hasGeometry ?wktCoordinate ?address
WHERE {
  ?listing a edintinm:RealEstateListing ;
           dcterms:identifier ?listingId ;
           edintinm:refersToCadastralEntity ?cadastralEntity .
  
  OPTIONAL {
    ?cadastralEntity geo:hasGeometry ?hasGeometry .
    
    OPTIONAL {
      ?hasGeometry geo:asWKT ?wktCoordinate .
    }
  }
  
  OPTIONAL {
    ?cadastralEntity catastro:hasAddress ?address .
  }
}

ORDER BY ?listingId
